```kira
function fakeDrawCubeTrailingAlias(frame: FakeGraphicsFrame, state: FakeCubeState) {
    let pass = FakeRenderPassDescriptor {
        label: "fake-3d-cube-trailing-alias"
        colorAttachments: [
            FakeColorAttachment {
                target: FakeRenderTarget {
                    kind: "swapchain"
                }
            }
        ]
    }
    let pipeline = state.pipeline
    let vertices = state.vertices
    let indexCount = state.indexCount

    frame.renderPass(pass) { encoder in
        encoder.setPipeline(pipeline)
        encoder.setVertexBuffer(vertices)
        encoder.drawIndexed(indexCount)
    }
    return
}
```

- The older explicit `beginRenderPass(...)`, helper encode call, and `endPass(...)` pattern remains the broadest safe fallback.
